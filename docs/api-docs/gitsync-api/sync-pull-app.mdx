<EndpointHeader method="PUT" path="/apps/{appId}" />

<ApiColumns>
<ApiLeft>

Sync and pull the latest changes from GitHub for an existing ToolJet app.

<ParamsTable title="Path Parameters" params={[{"name":"appId","type":"string","required":true,"description":"ID of the app.","in":"path"}]} />

<ParamsTable title="Query Parameters" params={[{"name":"createMode","type":"string","required":true,"description":"Must be set to git to enable git-sync mode for this operation.","in":"query"}]} />

<ErrorsTable errors={[
  {"status":"400","description":"Git provider not enabled or git push failed."},
  {"status":"401","description":"Invalid Authorization token."},
  {"status":"403","description":"Insufficient permissions."},
  {"status":"404","description":"App not found or not available on Community Edition."},
  {"status":"451","description":"GIT_SYNC license not active."}
]} />

</ApiLeft>
<ApiRight>

<RequestExample code={`curl -X PUT "https://{your-domain}/api/ext/apps/{appId}?createMode=<createMode>" \

  -H "Content-Type: application/json"`} />

**Response:** `200`

</ApiRight>
