<EndpointHeader method="PUT" path="/apps/{appId}" />

Sync and pull the latest changes from GitHub for an existing ToolJet app.

<ParamsTable title="Path Parameters" params={[{"name":"appId","type":"string","required":true,"description":"ID of the app.","in":"path"}]} />

<ParamsTable title="Query Parameters" params={[{"name":"createMode","type":"string","required":true,"description":"","in":"query"}]} />



<RequestExample code={`curl -X PUT "https://{your-domain}/api/ext/apps/{appId}?createMode=<createMode>" \
  -H "Authorization: Basic <access_token>" \
  -H "Content-Type: application/json"`} />

**Response:** `200`
