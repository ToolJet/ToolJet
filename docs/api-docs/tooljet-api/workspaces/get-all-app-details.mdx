<EndpointHeader method="GET" path="/workspace/{workspaceId}/apps" />

Returns app details for all applications in a workspace.

<ParamsTable title="Path Parameters" params={[{"name":"workspaceId","type":"string","required":true,"description":"","in":"path"}]} />





<RequestExample code={`curl -X GET "https://{your-domain}/api/ext/workspace/{workspaceId}/apps" \
  -H "Authorization: Basic <access_token>" \
  -H "Content-Type: application/json"`} />
