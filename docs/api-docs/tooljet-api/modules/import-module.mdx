<EndpointHeader method="POST" path="/import/workspace/{workspaceId}/modules" />

<ApiColumns>
<ApiLeft>

Imports a module into a workspace from a previously exported payload.

<ParamsTable title="Path Parameters" params={[{"name":"workspaceId","type":"string","required":true,"description":"The unique identifier of the workspace to import the module into.","in":"path"}]} />



<ParamsTable title="Request Body" params={[{"name":"tooljet_version","type":"string","required":true,"description":"The ToolJet version the module was exported from.","in":"body"},{"name":"app","type":"object","required":false,"description":"Module definition data from the export step.","in":"body"},{"name":"appName","type":"string","required":false,"description":"Overrides the module name on import.","in":"body"}]} />


<ErrorsTable errors={[
  {"status":"400","description":"workspaceId is not a valid UUID, workspace does not exist, payload tooljet_version is newer than the server, or definition type is not module."},
  {"status":"409","description":"Schema-level TJDB violation (e.g. duplicate column names) — table name collisions are handled by auto-renaming."},
  {"status":"451","description":"EXTERNAL_API license not active on the instance."},
  {"status":"500","description":"Unhandled error during import — message: Failed to import module."}
]} />

</ApiLeft>
<ApiRight>

<RequestExample code={`curl -X POST "https://{your-domain}/api/ext/import/workspace/{workspaceId}/modules" \\
    -H "Authorization: Basic <access_token>" \\
    -H "Content-Type: application/json" \\
    -d '{
      "tooljet_version": "3.16.0",
      "appName": "My Module"
    }'`} />

<ResponseExample statusCode="200" body={`{
  "message": "Module imported successfully."
}`} />

</ApiRight>
</ApiColumns>
