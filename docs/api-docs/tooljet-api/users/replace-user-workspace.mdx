<EndpointHeader method="PATCH" path="/user/{id}/workspace/{workspaceId}" />

<ParamsTable title="Request Body" params={[{"name":"id","type":"string","required":false,"description":"ID of the workspace (alternative to path parameter).","in":"body"},{"name":"name","type":"string","required":false,"description":"Name of the workspace.","in":"body"},{"name":"status","type":"string","required":false,"description":"User's status in this workspace. ]} />
