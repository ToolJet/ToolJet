---
id: replace-user-workspaces
title: "Replace User Relationship in Multiple Workspaces"
---

<EndpointHeader method="PUT" path="/user/{id}/workspaces" />

<ApiColumns>
<ApiLeft>

Updates a user's role, status, or group assignments across multiple workspaces in one call. This fully replaces the user's workspace memberships — any workspace not included in the request will be removed. Send an empty array to remove the user from all workspaces.


<ParamsTable title="Path Parameters" params={[{"name":"id","type":"string","required":true,"description":"UUID of the user.","in":"path"}]} />

<ParamsTable title="Request Body" params={[
  {"name":"id","type":"string","required":false,"description":"UUID of the workspace. Either id or name must be provided.","in":"body"},
  {"name":"name","type":"string","required":false,"description":"Name of the workspace. Either id or name must be provided.","in":"body"},
  {"name":"role","type":"string","required":true,"description":"User's role in the workspace. Accepted values: ADMIN, MEMBER.","in":"body"},
  {"name":"status","type":"string","required":true,"description":"User's status in the workspace. Accepted values: ACTIVE, ARCHIVED.","in":"body"},
  {"name":"groups","type":"array","required":false,"description":"List of groups to assign the user to in this workspace. Each item must have an id (group UUID).","in":"body"}
]} />

<ErrorsTable errors={[
  {"status":"400","description":"Workspace not found, user account inactive, group not found, or invalid group type."},
  {"status":"402","description":"User seat limit exceeded."},
  {"status":"403","description":"External API disabled, missing EXTERNAL_API license, or invalid Authorization token."},
  {"status":"404","description":"User not found."},
  {"status":"500","description":"Internal server error (e.g. invalid enum values in request body)."}
]} />

</ApiLeft>
<ApiRight>
